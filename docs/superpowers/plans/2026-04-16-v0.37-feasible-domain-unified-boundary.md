```python
from dataclasses import dataclass

import pandas as pd


@dataclass(frozen=True)
class FeasibleDomainBundle:
    weekly_bounds: pd.DataFrame
    hourly_bounds: pd.DataFrame
    settlement_semantics: pd.DataFrame


def _resolve_domain_config(config: dict[str, object]) -> dict[str, object]:
    if "policy_feasible_domain" in config:
        return dict(config["policy_feasible_domain"])
    projection = dict(config.get("policy_projection", {}))
    return {
        "enabled": True,
        "strict_mode": False,
        "bind_upper_actions": True,
        "bind_lower_actions": True,
        "bind_settlement_mode": True,
        "non_negative_position_required": bool(projection.get("enforce_non_negative_position", True)),
    }


def compile_feasible_domain(
    *,
    config: dict[str, object],
    policy_state_trace: pd.DataFrame,
    weekly_metadata: pd.DataFrame,
) -> FeasibleDomainBundle:
    domain_cfg = _resolve_domain_config(config)
    merged = weekly_metadata[["week_start", "forecast_weekly_net_demand_mwh"]].merge(policy_state_trace, on="week_start", how="left").fillna(0.0)
    weekly_rows: list[dict[str, object]] = []
    hourly_rows: list[dict[str, object]] = []
    semantic_rows: list[dict[str, object]] = []
    for row in merged.itertuples(index=False):
        forecast = float(row.forecast_weekly_net_demand_mwh)
        linked_active = float(getattr(row, "lt_price_linked_active", 0.0))
        ancillary_tight = float(getattr(row, "ancillary_freq_reserve_tight", 0.0))
        contract_adjustment_max = 0.18 * forecast if linked_active > 0.5 else 0.30 * forecast
        exposure_band_max = 0.12 * forecast if ancillary_tight > 0.5 else 0.20 * forecast
        settlement_mode = "linked_proxy" if linked_active > 0.5 else "legacy_proxy"
        reason_code = "linked_tight" if linked_active > 0.5 and ancillary_tight > 0.5 else "baseline"
        weekly_rows.append(
            {
                "week_start": pd.Timestamp(row.week_start),
                "contract_adjustment_min_mwh": -contract_adjustment_max,
                "contract_adjustment_max_mwh": contract_adjustment_max,
                "contract_position_min_mwh": 0.0 if domain_cfg["non_negative_position_required"] else -forecast,
                "contract_position_max_mwh": forecast,
                "exposure_band_min_mwh": 0.0,
                "exposure_band_max_mwh": exposure_band_max,
                "feasible_domain_triggered": float(linked_active > 0.5 or ancillary_tight > 0.5),
                "bound_reason_code": reason_code,
                "settlement_mode": settlement_mode,
            }
        )
        for hour_index in range(168):
            hourly_rows.append(
                {
                    "week_start": pd.Timestamp(row.week_start),
                    "hour_index": hour_index,
                    "spot_hedge_min_mwh": -exposure_band_max / 168.0,
                    "spot_hedge_max_mwh": exposure_band_max / 168.0,
                    "policy_shrink_factor": 0.55 if ancillary_tight > 0.5 else 1.0,
                    "bound_reason_code": reason_code,
                }
            )
        semantic_rows.append(
            {
                "week_start": pd.Timestamp(row.week_start),
                "settlement_mode": settlement_mode,
                "bound_reason_code": reason_code,
            }
        )
    return FeasibleDomainBundle(
        weekly_bounds=pd.DataFrame(weekly_rows),
        hourly_bounds=pd.DataFrame(hourly_rows),
        settlement_semantics=pd.DataFrame(semantic_rows),
    )
```

```python
def _encode_categories(series: pd.Series) -> tuple[torch.Tensor, dict[str, int]]:
    categories = {value: index for index, value in enumerate(sorted(series.fillna("unknown").astype(str).unique()))}
    encoded = torch.tensor(series.fillna("unknown").astype(str).map(categories).to_numpy(dtype="int64"), dtype=torch.int64)
    return encoded, categories
```
